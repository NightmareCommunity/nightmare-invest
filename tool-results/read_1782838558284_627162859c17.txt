  1102→function AumForecastChart({ navTrend, totalAum }: { navTrend: any[]; totalAum: number }) {
  1103→  const [showForecast] = useState(true);
  1104→
  1105→  // Generate forecast data using linear regression
  1106→  const { historicalData, forecastData } = useMemo(() => {
  1107→    if (!navTrend || navTrend.length === 0) {
  1108→      // Generate sample data
  1109→      const now = new Date();
  1110→      const hist = [];
  1111→      for (let i = 89; i >= 0; i--) {
  1112→        const d = new Date(now);
  1113→        d.setDate(d.getDate() - i);
  1114→        const baseAum = 45_000_000 + (90 - i) * 80_000;
  1115→        hist.push({
  1116→          date: d.toISOString().split("T")[0],
  1117→          aum: baseAum + Math.sin(i * 0.15) * 2_000_000 + (Math.random() - 0.5) * 1_500_000,
  1118→          type: "historical",
  1119→        });
  1120→      }
  1121→      return { historicalData: hist, forecastData: generateForecast(hist) };
  1122→    }
  1123→
  1124→    const hist = navTrend.map((p: any) => ({
  1125→      date: typeof p.date === "string" ? p.date.split("T")[0] : new Date(p.date).toISOString().split("T")[0],
  1126→      aum: p.aum ?? p.nav * 10000000 ?? 0,
  1127→      type: "historical" as const,
  1128→    }));
  1129→
  1130→    return { historicalData: hist, forecastData: generateForecast(hist) };
  1131→  }, [navTrend]);
  1132→
  1133→  function generateForecast(hist: { date: string; aum: number }[]) {
  1134→    if (hist.length < 2) return [];
  1135→
  1136→    // Simple linear regression
  1137→    const n = hist.length;
  1138→    const xMean = (n - 1) / 2;
  1139→    const yMean = hist.reduce((s, p) => s + p.aum, 0) / n;
  1140→    let num = 0;
  1141→    let den = 0;
  1142→    for (let i = 0; i < n; i++) {
  1143→      num += (i - xMean) * (hist[i].aum - yMean);
  1144→      den += (i - xMean) ** 2;
  1145→    }
  1146→    const slope = den !== 0 ? num / den : 0;
  1147→    const intercept = yMean - slope * xMean;
  1148→
  1149→    // Calculate standard error for confidence interval
  1150→    const residuals = hist.map((p, i) => p.aum - (slope * i + intercept));
  1151→    const se = Math.sqrt(residuals.reduce((s, r) => s + r * r, 0) / (n - 2));
  1152→
  1153→    const lastDate = new Date(hist[hist.length - 1].date);
  1154→    const forecast = [];
  1155→    for (let i = 1; i <= 90; i++) {
  1156→      const d = new Date(lastDate);
  1157→      d.setDate(d.getDate() + i);
  1158→      const predictedAum = slope * (n + i - 1) + intercept;
  1159→      const uncertainty = se * Math.sqrt(1 + 1 / n + ((n + i - 1 - xMean) ** 2) / den);
  1160→      forecast.push({
  1161→        date: d.toISOString().split("T")[0],
  1162→        aum: predictedAum,
  1163→        aumUpper: predictedAum + 1.96 * uncertainty,
  1164→        aumLower: predictedAum - 1.96 * uncertainty,
  1165→        type: "forecast",
  1166→      });
  1167→    }
  1168→    return forecast;
  1169→  }
  1170→
  1171→  const chartData = useMemo(() => {
  1172→    const combined = [...historicalData];
  1173→    if (showForecast && forecastData.length > 0) {
  1174→      // Add forecast data starting from last historical point
  1175→      combined.push(...forecastData);
  1176→    }
  1177→    return combined;
  1178→  }, [historicalData, forecastData, showForecast]);
  1179→
  1180→  return (
  1181→    <ResponsiveContainer width="100%" height="100%">
  1182→      <AreaChart data={chartData} margin={{ top: 8, right: 8, left: 0, bottom: 0 }}>
  1183→        <defs>
  1184→          <linearGradient id="forecastArea" x1="0" y1="0" x2="0" y2="1">
  1185→            <stop offset="0%" stopColor="#D4AF37" stopOpacity={0.25} />
  1186→            <stop offset="100%" stopColor="#D4AF37" stopOpacity={0} />
  1187→          </linearGradient>
  1188→          <linearGradient id="confidenceArea" x1="0" y1="0" x2="0" y2="1">
  1189→            <stop offset="0%" stopColor="#D4AF37" stopOpacity={0.08} />
  1190→            <stop offset="100%" stopColor="#D4AF37" stopOpacity={0.02} />
  1191→          </linearGradient>
  1192→        </defs>
  1193→        <CartesianGrid strokeDasharray="4 6" stroke="rgba(255,255,255,0.06)" vertical={false} />
  1194→        <XAxis
  1195→          dataKey="date"
  1196→          tickFormatter={(d) => new Date(d).toLocaleDateString("en-US", { month: "short", day: "numeric" })}
  1197→          tick={{ fill: "#8A8A8A", fontSize: 11 }}
  1198→          stroke="rgba(255,255,255,0.08)"
  1199→          minTickGap={40}
  1200→        />
  1201→        <YAxis
  1202→          tickFormatter={(v) => `$${(v / 1e6).toFixed(0)}M`}
  1203→          tick={{ fill: "#8A8A8A", fontSize: 11 }}
  1204→          stroke="rgba(255,255,255,0.08)"
  1205→          width={56}
  1206→        />
  1207→        <Tooltip

  1400→
  1401→function CapitalFlowPredictions({ depositVolume, withdrawalVolume }: { depositVolume: number; withdrawalVolume: number }) {
  1402→  const predictionData = useMemo(() => {
  1403→    // Generate 30-day predictions based on historical volumes
  1404→    const dailyDeposit = (depositVolume || 3_100_000) / 30;
  1405→    const dailyWithdrawal = (withdrawalVolume || 1_100_000) / 30;
  1406→    const days = [];
  1407→    for (let i = 1; i <= 30; i++) {
  1408→      const dayOfWeek = (new Date().getDay() + i) % 7;
  1409→      // Lower activity on weekends
  1410→      const weekendFactor = (dayOfWeek === 0 || dayOfWeek === 6) ? 0.3 : 1;
  1411→      const randomFactor = 0.7 + Math.random() * 0.6;
  1412→      days.push({
  1413→        day: i,
  1414→        label: `Day ${i}`,
  1415→        predictedDeposits: Math.round(dailyDeposit * weekendFactor * randomFactor),
  1416→        predictedWithdrawals: Math.round(dailyWithdrawal * weekendFactor * randomFactor),
  1417→        actualDeposits: i <= 7 ? Math.round(dailyDeposit * weekendFactor * (0.6 + Math.random() * 0.8)) : undefined,
  1418→        actualWithdrawals: i <= 7 ? Math.round(dailyWithdrawal * weekendFactor * (0.6 + Math.random() * 0.8)) : undefined,
  1419→      });
  1420→    }
  1421→    return days;
  1422→  }, [depositVolume, withdrawalVolume]);
  1423→
  1424→  // Weekly aggregation
  1425→  const weeklyData = useMemo(() => {
  1426→    const weeks: { week: string; predictedDeposits: number; predictedWithdrawals: number; actualDeposits: number; actualWithdrawals: number }[] = [];
  1427→    for (let w = 0; w < 4; w++) {
  1428→      const weekDays = predictionData.slice(w * 7, (w + 1) * 7);
  1429→      weeks.push({
  1430→        week: `Week ${w + 1}`,
  1431→        predictedDeposits: weekDays.reduce((s, d) => s + d.predictedDeposits, 0),
  1432→        predictedWithdrawals: weekDays.reduce((s, d) => s + d.predictedWithdrawals, 0),
  1433→        actualDeposits: weekDays.reduce((s, d) => s + (d.actualDeposits ?? 0), 0),
  1434→        actualWithdrawals: weekDays.reduce((s, d) => s + (d.actualWithdrawals ?? 0), 0),
  1435→      });
  1436→    }
  1437→    return weeks;
  1438→  }, [predictionData]);
  1439→
  1440→  const totalPredictedDeposits = predictionData.reduce((s, d) => s + d.predictedDeposits, 0);
  1441→  const totalPredictedWithdrawals = predictionData.reduce((s, d) => s + d.predictedWithdrawals, 0);
  1442→  const totalActualDeposits = predictionData.reduce((s, d) => s + (d.actualDeposits ?? 0), 0);
  1443→
  1444→  return (
  1445→    <div className="mt-4 space-y-4">
  1446→      {/* Summary row */}
  1447→      <div className="grid gap-3 sm:grid-cols-3">
  1448→        <div className="rounded-lg bg-profit/[0.06] p-3 text-center">
  1449→          <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Predicted Deposits</div>
  1450→          <div className="mt-1 font-metric text-lg font-bold text-profit">{fmtUSD(totalPredictedDeposits, { compact: true })}</div>
  1451→        </div>
  1452→        <div className="rounded-lg bg-loss/[0.06] p-3 text-center">
  1453→          <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Predicted Withdrawals</div>
  1454→          <div className="mt-1 font-metric text-lg font-bold text-loss">{fmtUSD(totalPredictedWithdrawals, { compact: true })}</div>
  1455→        </div>
  1456→        <div className="rounded-lg bg-gold/[0.06] p-3 text-center">
  1457→          <div className="text-[10px] uppercase tracking-wider text-muted-foreground">Net Forecast</div>
  1458→          <div className="mt-1 font-metric text-lg font-bold text-gold">{fmtUSD(totalPredictedDeposits - totalPredictedWithdrawals, { compact: true })}</div>
  1459→        </div>
  1460→      </div>
  1461→
  1462→      {/* Mini bar chart */}
  1463→      <div className="h-48">
  1464→        <ResponsiveContainer width="100%" height="100%">
  1465→          <BarChart data={weeklyData} margin={{ top: 8, right: 4, left: 4, bottom: 0 }}>
  1466→            <defs>
  1467→              <linearGradient id="predDepositBar" x1="0" y1="0" x2="0" y2="1">
  1468→                <stop offset="0%" stopColor="#00c896" stopOpacity={0.8} />
  1469→                <stop offset="100%" stopColor="#00c896" stopOpacity={0.4} />
  1470→              </linearGradient>
  1471→              <linearGradient id="predWithdrawalBar" x1="0" y1="0" x2="0" y2="1">
  1472→                <stop offset="0%" stopColor="#ff4d4f" stopOpacity={0.8} />
  1473→                <stop offset="100%" stopColor="#ff4d4f" stopOpacity={0.4} />
  1474→              </linearGradient>
  1475→            </defs>
  1476→            <CartesianGrid strokeDasharray="3 3" stroke="rgba(255,255,255,0.04)" />
  1477→            <XAxis dataKey="week" tick={{ fill: "#7A7A7A", fontSize: 11 }} stroke="rgba(255,255,255,0.06)" />
  1478→            <YAxis tickFormatter={(v) => `$${(v / 1e6).toFixed(1)}M`} tick={{ fill: "#7A7A7A", fontSize: 10 }} stroke="rgba(255,255,255,0.06)" width={48} />
  1479→            <Tooltip

  1493→      {/* Actual vs predicted indicator */}
